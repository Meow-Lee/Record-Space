## 문자열 슬라이스

```rust
let s = String::from("hello world");

// 만드는 방식은 String 참조자와 비슷하지만, [starting_index..ending_index] 형식
// ending_index에서 1을 뺀 위치까지 슬라이스를 생성
// 즉, &s[6..11]이면 s의 7번째 바이트를 가리키는 (1부터 시작해서) 포인터와
// 길이 값 5를 갖는 슬라이스가 됨
let hello = &s[0..5];
let world = &s[6..11];

// ..범위 표현법은 인덱스0부터 시작하는 경우 앞의 값을 생략할 수 있음
let slice = &s[0..2]
let slice = &s[..2]

// String 맨 마지막 바이트까지 포함하는 슬라이스는 뒤의 값을 생략할 수 있음
let len = s.len();

let slice = &s[3..len];
let slice = &s[3..];

// 슬라이스 반환 함수
// 문자열 슬라이스를 나타내는 타입은 &str로 작성
fn first_word(s: &String) -> &str {
	let bytes = s.as_bytes();
	for(i, &item) in bytes.iter().enumerate() {
		if item == b' '{
			return &s[0..i];
	}
	}
	&s[..]
}
```
